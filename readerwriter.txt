#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>

sem_t r, w;
int r_count = 0;


void* reader()
{
	sem_wait(&r);
		r_count++;
	 	if(r_count == 1)
	 	{
			sem_wait(&w);
		}
	sem_post(&r);
	printf("reader is reading\n");
	sem_wait(&r);
		r_count--;
		if(r_count == 0)
		{
			sem_post(&w);
		}
	sem_post(&r);
}

void* writer()
{
	sem_wait(&w);
		printf("writer is writing\n");
	sem_post(&w);
}
 
int main(){
	   
	    
	    pthread_t rtid[3];
	    pthread_t wtid[3];
	    
	    sem_init( &r, 0, 1);
	    sem_init( &w, 0, 1);
	    
	    for(int i = 0; i < 3 ; i++)
	    {
	    	pthread_create(&wtid[i], NULL, writer, NULL);
	    	pthread_create(&rtid[i], NULL, reader, NULL);
	    	
	    }
	    
	    for(int i = 0; i<3 ; i++)
	    {
	    	pthread_join(rtid[i], NULL);
	    	pthread_join(wtid[i], NULL);
	    }    
}
